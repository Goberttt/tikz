\newcommand{\hypercube}{
    \foreach\x in {-1,1} {
        \foreach\y in {-1,1} {
            \foreach\z in {-1,1} {
                \coordinate (\x\y\z-1) at (\x+\offsetx*\z, \y+\offsety*\z);
            }
        }
    }
    \foreach\x in {-1,1} {
        \foreach\y in {-1,1} {
            \foreach\z in {-1,1} {
                \coordinate (\x\y\z1) at (0.5\x+\offsetxx*\z, 0.5\y+\offsetyy*\z);
            }
        }
    }

    
    \foreach\a in {-1} {
        \foreach\b in {-1,1} {
            \foreach\c in {-1,1} {
                \foreach\d in {-1,1} {
                    \draw[very thin, \if\d 1{dashed}\fi] (\a\b\c\d) -- (1\b\c\d);
                }
            }
        }
    }
    \foreach\a in {-1,1} {
        \foreach\b in {-1} {
            \foreach\c in {-1,1} {
                \foreach\d in {-1,1} {
                    \draw[very thin, \if\d 1{dashed}\fi] (\a\b\c\d) -- (\a1\c\d);
                }
            }
        }
    }
    \foreach\a in {-1,1} {
        \foreach\b in {-1,1} {
            \foreach\c in {-1} {
                \foreach\d in {-1,1} {
                    \draw[very thin, \if\d 1{dashed}\fi] (\a\b\c\d) -- (\a\b1\d);
                }
            }
        }
    }
    \foreach\a in {-1,1} {
        \foreach\b in {-1,1} {
            \foreach\c in {-1,1} {
                \foreach\d in {-1} {
                    \draw[dashed] (\a\b\c\d) -- (\a\b\c1);
                }
            }
        }
    }
}
