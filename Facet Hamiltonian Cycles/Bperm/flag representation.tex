\newcommand{\offsetx}{0.4}
\newcommand{\offsety}{0.6}

\newcommand{\cube}{
    \foreach\x in {0,1} {
        \foreach\y in {0,1} {
            \foreach\z in {0,1} {
                \coordinate (\x\y\z) at (\x+\offsetx*\z, \y+\offsety*\z);
            }
        }
    }
    \foreach\x in {0} {
        \foreach\y in {0,1} {
            \foreach\z in {0,1} {
                \draw[very thin] (\x\y\z) -- (1\y\z);
            }
        }
    }
    \foreach\x in {0,1} {
        \foreach\y in {0} {
            \foreach\z in {0,1} {
                \draw[very thin] (\x\y\z) -- (\x1\z);
            }
        }
    }
    \foreach\x in {0,1} {
        \foreach\y in {0,1} {
            \foreach\z in {0} {
                \draw[very thin] (\x\y\z) -- (\x\y1);
            }
        }
    }
}
